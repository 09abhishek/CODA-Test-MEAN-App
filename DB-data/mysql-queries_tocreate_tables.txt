
-- create users Table

CREATE TABLE `users` (
  `user_id` int(11) NOT NULL AUTO_INCREMENT,
  `user_name` varchar(45) NOT NULL,
  `name` varchar(45) NOT NULL,
  `user_pwd` varchar(45) NOT NULL,
  `email` varchar(45) NOT NULL,
  `user_role` varchar(45) NOT NULL,
  `age` varchar(45) DEFAULT NULL,
  `sex` varchar(45) DEFAULT NULL,
  `phone_number` varchar(45) DEFAULT NULL,
  PRIMARY KEY (`user_id`)
) ENGINE=InnoDB AUTO_INCREMENT=106 DEFAULT CHARSET=latin1


--create movies Table

CREATE TABLE `movies` (
  `movie_id` int(11) NOT NULL AUTO_INCREMENT,
  `movie_name` varchar(45) NOT NULL,
  `movie_title` varchar(45) NOT NULL,
  `avg_ratings` int(11) NOT NULL,
  `rating` int(11) NOT NULL,
  `movie_description` varchar(45) DEFAULT NULL,
  `language` varchar(45) DEFAULT NULL,
  `cast_crew` varchar(45) DEFAULT NULL,
  `genre` varchar(45) DEFAULT NULL,
  `release_date` varchar(45) DEFAULT NULL,
  `image_url` varchar(200) DEFAULT NULL,
  PRIMARY KEY (`movie_id`)
) ENGINE=InnoDB AUTO_INCREMENT=10014 DEFAULT CHARSET=latin1

--create comments Table

CREATE TABLE `comments` (
  `comment_id` int(11) NOT NULL AUTO_INCREMENT,
  `user_id` varchar(45) NOT NULL,
  `movie_id` varchar(45) NOT NULL,
  `user_name` varchar(45) NOT NULL,
  `star_rating` varchar(45) NOT NULL,
  `description` varchar(45) DEFAULT NULL,
  `disliked_count` varchar(45) DEFAULT NULL,
  `liked_count` varchar(45) DEFAULT NULL,
  PRIMARY KEY (`comment_id`)
) ENGINE=InnoDB AUTO_INCREMENT=5023 DEFAULT CHARSET=latin1
